// Sheet ที่ต้องสร้าง: "Applicants", "Branches", "Holidays" function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); if (data.type === 'branches') { var s = ss.getSheetByName('Branches'); s.clearContents(); s.appendRow(['name','wdOpen','wdClose','weOpen','weClose','weekendOpen','status','openDate']); data.rows.forEach(r => s.appendRow(r)); } else if (data.type === 'holidays') { var s = ss.getSheetByName('Holidays'); s.clearContents(); s.appendRow(['date','name']); data.rows.forEach(r => s.appendRow(r)); } else { var s = ss.getSheetByName('Applicants'); s.appendRow([new Date(), data.fullname, data.nickname, data.birthdate, data.age, data.nationality, data.religion, data.phone, data.email, data.address, data.position, data.source, data.sourceDetail, data.trainBranch, data.applyBranch, data.trainStart, data.trainEnd, data.eduLevel, data.school, data.experience, data.referrer, data.applicantNote, data.status, data.trainBranchNote]); } return ContentService.createTextOutput(JSON.stringify({result:'ok'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var type = e.parameter.type; if (type === 'branches') { var data = ss.getSheetByName('Branches').getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } if (type === 'holidays') { var data = ss.getSheetByName('Holidays').getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } }
